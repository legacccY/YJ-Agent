### 实验 1：QCTS

```python
import torch
import torch.nn as nn
from scipy.optimize import minimize

class QCTS(nn.Module):
    def __init__(self):
        super().__init__()
        self.T0 = nn.Parameter(torch.tensor(0.0))
        self.alpha = nn.Parameter(torch.tensor(0.0))
    
    def forward(self, logits, q_bar):
        T = torch.nn.functional.softplus(self.T0 + self.alpha * (1 - q_bar))
        return logits / T.unsqueeze(-1)

# 加载 Std VIB 在验证集上的 logits 和对应的 q_bar
# logits_val: [N, 2], q_bar_val: [N]
# labels_val: [N]

model = QCTS()
optimizer = torch.optim.LBFGS([model.T0, model.alpha], lr=0.01, max_iter=200)

def closure():
    optimizer.zero_grad()
    calibrated = model(logits_val, q_bar_val)
    loss = nn.CrossEntropyLoss()(calibrated, labels_val)
    loss.backward()
    return loss

optimizer.step(closure)
print(f"T0={model.T0.item():.4f}, alpha={model.alpha.item():.4f}")
print(f"T(q=1.0)={torch.nn.functional.softplus(model.T0).item():.4f}")
print(f"T(q=0.0)={torch.nn.functional.softplus(model.T0 + model.alpha).item():.4f}")
```
